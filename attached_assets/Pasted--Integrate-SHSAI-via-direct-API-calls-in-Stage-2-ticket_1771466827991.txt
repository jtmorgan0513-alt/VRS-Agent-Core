javascriptconst SHSAI_BASE_URL = 'https://ais.tellurideplatform.com/tell/api/hs/routing/v1';

// Generate random track ID
function generateTrackId() {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  let result = '';
  for (let i = 0; i < 32; i++) {
    result += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return result;
}

// Step 1: Initialize session
async function initSession(agentLdapId) {
  const trackId = generateTrackId();
  
  const response = await fetch(`${SHSAI_BASE_URL}/init`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/json'
    },
    body: JSON.stringify({
      chat_bot_type: 'HS_ROUTE_ASSISTANT',
      trackId: trackId,
      parameters: {
        userId: agentLdapId,
        techId: ''
      }
    })
  });
  
  const data = await response.json();
  // Extract sessionId and thread_id from response
  return {
    trackId,
    sessionId: response.headers.get('sessionid') || data.sessionId,
    threadId: data.thread_id || data.threadId
  };
}

// Step 2: Send prompt
async function sendPrompt(sessionId, trackId, threadId, promptText) {
  const response = await fetch(`${SHSAI_BASE_URL}/prompt`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
      'Sessionid': sessionId
    },
    body: JSON.stringify({
      chat_bot_type: 'HS_ROUTE_ASSISTANT',
      trackId: trackId,
      thread_id: threadId,
      parameters: {
        userMessage: promptText
      }
    })
  });
  
  return await response.json();
}

// Combined: Init + Query service order
async function queryServiceOrder(agentLdapId, serviceOrder) {
  // Initialize session
  const session = await initSession(agentLdapId);
  
  // Send service order query
  const prompt = `Give me all orders for customer having sample service order number ${serviceOrder}`;
  const result = await sendPrompt(session.sessionId, session.trackId, session.threadId, prompt);
  
  return {
    session,
    result
  };
}

module.exports = { initSession, sendPrompt, queryServiceOrder };
2. Create API Endpoint — /server/routes/shsai.js:
javascriptrouter.post('/api/shsai/query', authMiddleware, async (req, res) => {
  try {
    const { serviceOrder } = req.body;
    const agentLdapId = req.user.ldap_id; // From JWT session
    
    const result = await queryServiceOrder(agentLdapId, serviceOrder);
    
    res.json({
      success: true,
      data: result
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: 'Failed to query SHSAI'
    });
  }
});
