1. Create SHSAI Service — /server/services/shsai.js:
javascriptconst SHSAI_BASE_URL = 'https://ais.tellurideplatform.com/tell/api/hs/routing/v1';

// Generate random track ID (32 alphanumeric chars)
function generateTrackId() {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  let result = '';
  for (let i = 0; i < 32; i++) {
    result += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return result;
}

// Generate device info header
function generateDeviceInfo() {
  const uuid = crypto.randomUUID();
  return `${uuid}~~Server~~nodejs~~VRS~~1`;
}

// Step 1: Initialize session with LDAP
async function initSession(agentLdapId) {
  const trackId = generateTrackId();
  const deviceInfo = generateDeviceInfo();
  
  const response = await fetch(`${SHSAI_BASE_URL}/init`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/json, text/plain, */*',
      'deviceinfo': deviceInfo
    },
    body: JSON.stringify({
      chat_bot_type: 'HS_ROUTE_ASSISTANT',
      trackId: trackId,
      parameters: {
        userId: agentLdapId,
        techId: ''
      }
    })
  });
  
  // Get sessionId from response headers or body
  const sessionId = response.headers.get('sessionid');
  const data = await response.json();
  
  return {
    trackId,
    sessionId: sessionId || data.sessionId,
    threadId: data.thread_id || data.threadId,
    deviceInfo
  };
}

// Step 2: Send prompt and get SSE streamed response
async function queryServiceOrder(agentLdapId, serviceOrder) {
  // Initialize session
  const session = await initSession(agentLdapId);
  
  // Build the prompt
  const prompt = `Give me all orders for customer having sample service order number ${serviceOrder}`;
  
  // Call assist endpoint with SSE
  const response = await fetch(`${SHSAI_BASE_URL}/sse/assist`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'text/event-stream',
      'deviceinfo': session.deviceInfo,
      'sessionid': session.sessionId
    },
    body: JSON.stringify({
      chat_bot_type: 'HS_ROUTE_ASSISTANT',
      trackId: session.trackId,
      thread_id: session.threadId,
      parameters: {
        userMessage: prompt
      }
    })
  });
  
  // Read the SSE stream
  const text = await response.text();
  
  // Parse newline-delimited JSON chunks
  const chunks = text.split('\n').filter(line => line.trim());
  let fullContent = '';
  
  for (const chunk of chunks) {
    try {
      const parsed = JSON.parse(chunk);
      if (parsed.type === 'TEXT') {
        fullContent += parsed.content;
      }
      // Stop when complete
      if (parsed.type === 'STATUS' && parsed.content === 'COMPLETE') {
        break;
      }
    } catch (e) {
      // Skip non-JSON lines
    }
  }
  
  return {
    success: true,
    content: fullContent,
    session
  };
}

module.exports = { initSession, queryServiceOrder };
2. Create API Route — /server/routes/shsai.js:
javascriptconst express = require('express');
const router = express.Router();
const { queryServiceOrder } = require('../services/shsai');
const authMiddleware = require('../middleware/auth');

router.post('/api/shsai/query', authMiddleware, async (req, res) => {
  try {
    const { serviceOrder } = req.body;
    const agentLdapId = req.user.ldap_id;
    
    if (!serviceOrder) {
      return res.status(400).json({ success: false, error: 'Service order required' });
    }
    
    const result = await queryServiceOrder(agentLdapId, serviceOrder);
    
    res.json(result);
  } catch (error) {
    console.error('SHSAI query error:', error);
    res.status(500).json({ 
      success: false, 
      error: 'Failed to query service history' 
    });
  }
});

module.exports = router;
